void maxheap_down(vector<int>&v,int start,int end){
    int c=start;
    int l=2*c+1;
    int temp=v[c];

    }
}

void quickSort(vector<int>&v,int begin,int end){
    if(begin>=end-1)return ;
    int L=begin,R=end-1;
    int index=v[L];

    while(L<R){
        while(L<R){
            if(v[R]<index){
                v[L++]=v[R];
                break;
            }
            --R;
        }
        while(L<R){
            if(v[L]>index){
                v[R--]=v[L];
                break;
            }
            ++L;
        }
    }
    v[L]=index;
    quickSort(v,begin,L);
    quickSort(v,R+1,end);
}
