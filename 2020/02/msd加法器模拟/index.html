#include<cstdio>
#include<string>
#include<cmath>
#include<iostream>
using namespace std;
char T[4][4]={{'0','1','2'},{'1','1','0'},{'2','0','2'}};
char W[4][4]={{'0','2','1'},{'2','0','0'},{'1','0','0'}};
char T2[4][4]={{'0','0','0'},{'0','1','0'},{'0','0','2'}};
char W2[4][4]={{'0','1','2'},{'1','0','0'},{'2','0','0'}};
string MSD_add(string a , string b){
    string u1,u2,w1,w2,ans,zero="0";
    // 补零 
    int L = a.size()-b.size();  // 两数的位数差 
    if(L<0) for(int i = 0 ; i < -L ; i++) a.insert(0,zero);
    if(L>0) for(int i = 0 ; i < L ; i++) b.insert(0,zero);

    for(int i = 0 ; i < a.size() ; i++){
        u1.push_back(T[b[i]-'0'][a[i]-'0']);    //T1变换
        w1.push_back(W[b[i]-'0'][a[i]-'0']);    //W1变换
    }
    cout<<"T1变换: ";
    for(int i = 0 ; i < u1.size() ; i++)cout<<u1[i];
    cout<<"\nW1变换: ";
    for(int i = 0 ; i < w1.size() ; i++)cout<<w1[i];

    u1.push_back('0');  //u1后面补零 
    w1.insert(0,zero);  //w1前面补零 

    for(int i = 0 ; i < u1.size() ; i++){
        u2.push_back(T2[w1[i]-'0'][u1[i]-'0']); //T2变换 
        w2.push_back(W2[w1[i]-'0'][u1[i]-'0']);//W2变换 
    }

    cout<<"\nT2变换: ";
    for(int i = 0 ; i < u2.size() ; i++)cout<<u2[i];
    cout<<"\nW2变换: ";
    for(int i = 0 ; i < w2.size() ; i++)cout<<w2[i];

    u2.push_back('0');  //u2后面补零 
    w2.insert(0,zero);

    for(int i = 0 ; i < u2.size() ; i++)    // u2与w2变换 
        ans.push_back(T[w2[i]-'0'][u2[i]-'0']);

    cout<<"\nT变换: ";
    for(int i = 0 ; i < ans.size() ; i++)cout<<ans[i];

    long long sum = 0;
    for(int i = 0 ; i < ans.size() ; i++){  //求和 
        if(ans[i]!='2') sum += (ans[i]-'0') * pow(2,ans.size()-i-1);
        else sum -= pow(2,ans.size()-i-1);
    }
    printf("\n结果:%lld\n",sum);
}
int main(){
    string numa,numb;

    cin>>numa>>numb;    //以二进制输入 
    MSD_add(numa,numb);

    return 0;
}
/*

    0  1  2

        a
    0  1  2

        a
    0  1  2

b 1  0  1  0
2  0  0  2

        a
    0  1  2 
0  0  1  2
